    \section{Definition}

    \begin{equation}
        F(s) = \mathscr{L}\{f(t)\} = \int_0^\infty{e^{-st} f(t) dt}
    \end{equation}

    \begin{table}[h]
        \centering

            Multiplying by a Constant & af(at) & aF(s) \\
            Time Scaling & f(at) & \frac{1}{a}F\left(\frac{s}{a}\right) \\
